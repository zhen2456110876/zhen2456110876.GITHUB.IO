document.getElementById('copyWeatherInfo').addEventListener('click', () => { const selectedCityInfo = document.getElementById('selectedCityInfo'); // 使用兼容性写法尝试复制 const copyToClipboardFallback = () => { const tempTextarea = document.createElement('textarea'); tempTextarea.style.position = 'fixed'; tempTextarea.style.top = 0; tempTextarea.style.left = 0; tempTextarea.style.width = '1px'; tempTextarea.style.height = '1px'; tempTextarea.style.padding = 0; tempTextarea.style.border = 'none'; tempTextarea.style.outline = 'none'; tempTextarea.style.boxShadow = 'none'; tempTextarea.style.background = 'transparent'; tempTextarea.value = selectedCityInfo.innerText; document.body.appendChild(tempTextarea); tempTextarea.select(); try { document.execCommand('copy'); alert('天气信息已复制到剪贴板'); } catch (err) { alert('复制失败，请长按文字手动复制'); } document.body.removeChild(tempTextarea); }; if (navigator.clipboard) { navigator.clipboard.writeText(selectedCityInfo.innerText) .then(() => { alert('天气信息已复制到剪贴板'); }) .catch(copyToClipboardFallback); } else { copyToClipboardFallback(); } });
